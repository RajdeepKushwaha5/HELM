WITH recent_pulls AS (
  SELECT
    pr_number,
    pr_title,
    author,
    merged_at
  FROM github.pulls
  WHERE merged_at >= NOW() - INTERVAL '7 days'
),
recent_errors AS (
  SELECT
    title,
    project,
    level,
    times_seen,
    first_seen
  FROM sentry.issues
  WHERE first_seen >= NOW() - INTERVAL '7 days'
)
SELECT
  p.pr_number,
  p.pr_title,
  e.title      AS error_title,
  e.times_seen,
  e.level,
  e.first_seen
FROM recent_pulls p
JOIN recent_errors e
  ON e.first_seen BETWEEN p.merged_at
     AND p.merged_at + INTERVAL '2 hours'
ORDER BY e.times_seen DESC
